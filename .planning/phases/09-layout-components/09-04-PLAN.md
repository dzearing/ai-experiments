**Type definitions:**
```typescript
export type ContentLayoutMaxWidth = 'sm' | 'md' | 'lg' | 'xl' | 'full';
export type ContentLayoutPadding = 'none' | 'sm' | 'md' | 'lg';

export interface ContentLayoutProps {
  /** Header content (typically PageHeader) */
  header?: ReactNode;
  /** Main content */
  children: ReactNode;
  /** Footer content */
  footer?: ReactNode;
  /** Maximum content width */
  maxWidth?: ContentLayoutMaxWidth;
  /** Content padding */
  padding?: ContentLayoutPadding;
  /** Additional className */
  className?: string;
}
```

**Component structure:**
```tsx
<div className={styles.layout}>
  {header && <div className={styles.header}>{header}</div>}
  <main className={`${styles.content} ${styles[`maxWidth-${maxWidth}`]} ${styles[`padding-${padding}`]}`}>
    {children}
  </main>
  {footer && <footer className={styles.footer}>{footer}</footer>}
</div>
```

**CSS module (ContentLayout.module.css):**

Layout:
- .layout: min-height: 100%, display: flex, flex-direction: column
- .header: flex-shrink: 0 (stays at top)
- .content: flex: 1 (fills available space), margin: 0 auto for centering
- .footer: flex-shrink: 0 (stays at bottom)

MaxWidth variants:
- .maxWidth-sm: 640px
- .maxWidth-md: 768px
- .maxWidth-lg: 1024px
- .maxWidth-xl: 1280px
- .maxWidth-full: 100%

Padding variants:
- .padding-none: 0
- .padding-sm: var(--spacing-small10) or 8px
- .padding-md: var(--spacing) or 16px
- .padding-lg: var(--spacing-large10) or 24px

Use design tokens for spacing. Add isolation: isolate to establish stacking context.

Default values: maxWidth='lg', padding='md'

**LayoutDemo.tsx structure:**
```tsx
import { useState } from 'react';
import { TitleBar, PageHeader, SidePanel, ContentLayout, Button, IconButton } from '@claude-flow/ui-kit-react';

export function LayoutDemo() {
  const [activeTab, setActiveTab] = useState('work');
  const [sidePanelOpen, setSidePanelOpen] = useState(false);

  return (
    <div className={styles.app}>
      {/* App-level TitleBar */}
      <TitleBar
        logo={<AppLogo />}
        title="Coworker"
        tabs={[
          { value: 'work', label: 'Work' },
          { value: 'web', label: 'Web' },
        ]}
        activeTab={activeTab}
        onTabChange={setActiveTab}
        actions={<ProfileButton />}
      />

      {/* Main area with optional SidePanel */}
      <div className={styles.mainArea}>
        <SidePanel
          open={sidePanelOpen}
          onClose={() => setSidePanelOpen(false)}
          mode="push"
          position="left"
          header={<SidePanelHeader />}
        >
          <NavigationList />
        </SidePanel>

        <ContentLayout
          header={
            <PageHeader
              title="Dashboard"
              breadcrumbs={[
                { label: 'Home', href: '/' },
                { label: 'Dashboard' }
              ]}
              actions={<Button onClick={() => setSidePanelOpen(true)}>Toggle Panel</Button>}
            />
          }
        >
          <DemoContent />
        </ContentLayout>
      </div>
    </div>
  );
}
```
